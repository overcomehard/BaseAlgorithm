# 类中包含指针变量

- 问题描述: 类中定一个了一个指针变量
    - 参考: CV::Mat等
    - 其中就定义了指针变量: ```uchar* data```
    
    ```C++
    class ClassA
    {
    public:
        ClassA(int len = 0)
        {
            if (len > 0) {
                length = len;
                data = new int[length];
                for (int i = 0; i < length; i++)
                {
                    data[i] = 1;
                }
            }
            std::cout << "ClassA" << std::endl;
        }
        ~ClassA()
        {
            std::cout << "~ClassA" << std::endl;
        }
    
        int length = 0;
        int* data = NULL;
    };
    ```

- 这会引出很多问题
    - 如果析构函数中没有释放该变量会如何?
        - 显而易见, 容易造成内存泄漏
        - 因此类的定义修改如下:

        ```C++
        class ClassA
        {
        public:
            ClassA(int len = 0)
            {
                if (len > 0) {
                    length = len;
                    data = new int[length];
                    for (int i = 0; i < length; i++)
                    {
                        data[i] = 1;
                    }
                }
                std::cout << "ClassA" << std::endl;
            }
            ~ClassA()
            {
                if (data != NULL) {
                    delete data;
                    length = 0;
                    data = NULL;
                }
                std::cout << "~ClassA" << std::endl;
            }
        
            int length = 0;
            int* data = NULL;
        };
        ```
        
    - 如果没有定义```拷贝构造/赋值构造```会如何?
        - 假设析构函数中没有定义释放指针变量
            - 但是在传参, 赋值等操作时, 如果对指针内容做了修改, 所有对应的变量均会被修改
            
            ```C++
            #include <iostream>

            class ClassA
            {
            public:
                ClassA(int len = 0)
                {
                    if (len > 0) {
                        length = len;
                        data = new int[length];
                        for (int i = 0; i < length; i++)
                        {
                            data[i] = 1;
                        }
                    }
                    std::cout << "ClassA" << std::endl;
                }
                ~ClassA()
                {
                    std::cout << "~ClassA" << std::endl;
                }
            
                int length = 0;
                int* data = NULL;
            };

            void doA(ClassA a)
            {
                a.data[1] = 2;
                a.data[2] = 2;
            }

            int main()
            {
                ClassA a(4);  // 构造, 输出 ClassA
            
                std::cout << "before doA data a: ";
                for (int i = 0; i < a.length; i++)
                    std::cout << a.data[i] << ",";
                std::cout << std::endl;
                doA(a);  // 参数传递(调用默认拷贝构造), 运行结束后, 会析构函数里面的形参, 输出 ~ClassA
                std::cout << "after doA data a: ";
                for (int i = 0; i < a.length; i++)
                    std::cout << a.data[i] << ",";
                std::cout << std::endl;
            
                ClassA b;  // 构造, 输出 ClassA
                b = a;  // 赋值构造(调用默认的赋值构造)
                std::cout << "before change b, data a: ";
                for (int i = 0; i < a.length; i++)
                    std::cout << a.data[i] << ",";
                std::cout << std::endl;
                b.data[0] = 10;
                b.data[3] = 10;
                std::cout << "after change b, data a: ";
                for (int i = 0; i < a.length; i++)
                    std::cout << a.data[i] << ",";
                std::cout << std::endl;
                system("pause");
                return 0;
            }
            ```

            - 输出信息:
            
                ```
                ClassA
                before doA data a: 1,1,1,1,
                ~ClassA
                after doA data a: 1,2,2,1,
                ClassA
                before change b, data a: 1,2,2,1,
                after change b, data a: 10,2,2,10,
                ```
            
        - 假设析构函数中定义了释放指针变量的操作
            - 当该对象被赋值给别的变量时, 或者传参数时, 由于多次调用析构函数, 会报错
            - 为什么报错? 指针嘛, 都是相同的一段内存空间, 被多次释放, 肯定会报错
    - 拷贝构造用在什么地方?
        - 参数传递
        - 函数返回: 接受返回值的对象未被初始化
    - 赋值构造用在什么地方?
        - 赋值操作: ```=```
        - 函数返回: 接受返回值的对象已被初始化

```C++
#include <iostream>

class ClassA
{
public:
    ClassA(int len = 0)
    {
        if (len > 0) {
            length = len;
            data = new int[length];
            for (int i = 0; i < length; i++)
            {
                data[i] = 1;
            }
        }
        std::cout << "ClassA" << std::endl;
    }
    ~ClassA()
    {
        if (data != NULL) {
            delete data;
            length = 0;
            data = NULL;
        }
        std::cout << "~ClassA" << std::endl;
    }

    int length = 0;
    int* data = NULL;

    // 拷贝构造
    ClassA(const ClassA& t)
    {
        std::cout << "Copy" << std::endl;
        length = t.length;
        data = new int[t.length];
        for (int i = 0; i < t.length; i++) {
            data[i] = t.data[i];
        }
    }
    // 赋值构造
    inline ClassA& operator= (const ClassA &t)
    {
        std::cout << "Operator =" << std::endl;
        length = t.length;
        data = new int[t.length];
        for (int i = 0; i < t.length; i++) {
            data[i] = t.data[i];
        }
        return *this;
    }
};

void doA(ClassA a)
{
    a.data[1] = 2;
    a.data[2] = 2;
}

ClassA doB(ClassA a)
{
    a.data[0] = 9;
    return a;
}

int main()
{
    ClassA a(4);  // 构造, 输出: ClassA

    std::cout << "before doA data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;
    doA(a);  // 参数传递&形参析构, 输出: Copy, ~ClassA
    std::cout << "after doA data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;

    std::cout << "before doB data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;
    // 参数传递&拷贝构造&形参析构, 输出: Copy, Copy, ~ClassA
    // 这里为什么是"拷贝构造", 不是赋值构造?
    // 在声明变量b时, 并没有实例化, 因此就没有this指针
    ClassA b = doB(a);
    std::cout << "after doB data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;
    std::cout << "after doB data b: ";
    for (int i = 0; i < b.length; i++)
        std::cout << b.data[i] << ",";
    std::cout << std::endl;

    ClassA c;  // 构造, 输出: ClassA
    c = a;  // 赋值构造, 输出: Operator =
    std::cout << "before change c, data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;
    std::cout << "before change c, data c: ";
    for (int i = 0; i < c.length; i++)
        std::cout << c.data[i] << ",";
    std::cout << std::endl;
    c.data[0] = 10;
    c.data[3] = 10;
    std::cout << "after change c, data a: ";
    for (int i = 0; i < a.length; i++)
        std::cout << a.data[i] << ",";
    std::cout << std::endl;
    std::cout << "after change c, data c: ";
    for (int i = 0; i < c.length; i++)
        std::cout << c.data[i] << ",";
    std::cout << std::endl;

	std::cout << "before doB data a: ";
	for (int i = 0; i < a.length; i++)
		std::cout << a.data[i] << ",";
	std::cout << std::endl;
    // 这里的输出信息比较多: Copy, Copy, ~ClassA, Operator =, ~ClassA
    // 为什么会是这些?
    // 形参构造, 输出: Copy
    // 函数返回临时变量构造, 输出: Copy
    // 析构形参, 输出: ~ClassA
    // 将函数返回的临时变量赋值构造给变量c, 输出: Operator =
    // 析构函数返回的临时变量, 输出: ~ClassA
    // 与之前的函数返回差别在于, 前者是拷贝, 后者是赋值
	c = doB(a);
	std::cout << "after doB data a: ";
	for (int i = 0; i < a.length; i++)
		std::cout << a.data[i] << ",";
	std::cout << std::endl;
	std::cout << "after doB data c: ";
	for (int i = 0; i < c.length; i++)
		std::cout << c.data[i] << ",";
	std::cout << std::endl;
    system("pause");
    return 0;
}
```

    Copy
